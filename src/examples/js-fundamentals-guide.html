What is JavaScript?

JavaScript is the programming language of the web. It brings your HTML and CSS to life by adding interactivity, dynamic content updates, and user engagement features to websites.

JavaScript Syntax & Structure

JavaScript code can be embedded in HTML or written in separate files. Here's the basic syntax:

// Variables store data let message = "Hello, World!"; const number = 42; // Functions perform actions function greetUser(name) { return "Hello, " + name + "!"; } // Call the function console.log(greetUser("Student"));
  • Variables: Store and manipulate data (let, const, var)
  • Functions: Reusable blocks of code that perform specific tasks
  • Console: Tool for debugging and displaying output
  • Comments: Explanatory notes in your code (// or /* */)

Variables & Data Types

JavaScript Data Types

// Strings (text)
let firstName = "John";
let lastName = 'Doe';

// Numbers
let age = 25;
let price = 19.99;

// Booleans (true/false)
let isStudent = true;
let isGraduated = false;

// Arrays (lists)
let colors = ["red", "blue", "green"];
let numbers = [1, 2, 3, 4, 5];

// Objects
let person = {
    name: "Alice",
    age: 30,
    city: "New York"
};

Variable Declaration

// let - can be changed

Functions

Functions are the building blocks of JavaScript programs. They allow you to organize and reuse code.

// Function declaration function calculateArea(width, height) { let area = width * height; return area; } // Arrow function (modern syntax) const multiply = (a, b) => a * b; // Function with default parameters function greet(name = "World") { return `Hello, ${name}!`; } // Using functions let roomArea = calculateArea(10, 12); let result = multiply(5, 3); console.log(greet("Alice"));

Control Flow

Conditional Statements

// if/else statements
let score = 85;

if (score >= 90) {
    console.log("A grade");
} else if (score >= 80) {
    console.log("B grade");
} else if (score >= 70) {
    console.log("C grade");
} else {
    console.log("Study harder");
}

// Ternary operator (shorthand)
let status = age >= 18 ? "adult" : "minor";

// Switch statement
let day = "Monday";
switch (day) {
    case "Monday":
        console.log("Start of work week");
        break;
    case "Friday":
        console.log("TGIF!");
        break;
    default:
        console.log("Regular day");
}

Loops

// for loop
for (let i = 0; i < 5; i++) {
    console.log("Count: " + i);
}

// while loop
let count = 0;
while (count < 3) {
    console.log("While: " + count);
    count++;
}

// for...of loop (arrays)
let fruits = ["apple", "banana", "orange"];
for (let fruit of fruits) {
    console.log(fruit);
}

// forEach method
fruits.forEach(function(fruit, index) {
    console.log(index + ": " + fruit);
});

DOM Manipulation

The Document Object Model (DOM) allows JavaScript to interact with HTML elements on your webpage.

<html> <head> <title>My Page</title> </head> <body> <h1 id="main-title">Welcome</h1> <p class="description">This is content</p> <button id="click-me">Click Me</button> </body> </html>
// Selecting elements const title = document.getElementById('main-title'); const description = document.querySelector('.description'); const button = document.querySelector('#click-me'); // Changing content title.textContent = "JavaScript is Awesome!"; description.innerHTML = "<strong>Updated</strong> content"; // Changing styles title.style.color = "blue"; title.style.fontSize = "2rem"; // Adding/removing classes description.classList.add("highlight"); description.classList.remove("old-style");

Event Handling

Adding Event Listeners

// Click events
button.addEventListener('click', function() {
    alert('Button was clicked!');
});

// Arrow function version
button.addEventListener('click', () => {
    console.log('Button clicked!');
});

// Form events
const form = document.querySelector('form');
form.addEventListener('submit', function(event) {
    event.preventDefault(); // Stop form submission
    console.log('Form submitted');
});

// Keyboard events
document.addEventListener('keydown', function(event) {
    if (event.key === 'Enter') {
        console.log('Enter key pressed');
    }
});

Common JavaScript Patterns

Form Validation

function validateForm() {
    const email = document.getElementById('email').value;
    const password = document.getElementById('password').value;
    
    // Check if email is valid
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(email)) {
        showError('Please enter a valid email');
        return false;
    }
    
    // Check password length
    if (password.length < 8) {
        showError('Password must be at least 8 characters');
        return false;
    }
    
    return true;
}

function showError(message) {
    const errorDiv = document.getElementById('error');
    errorDiv.textContent = message;
    errorDiv.style.color = 'red';
}

API Requests (Fetch)

// Fetch data from an API
async function fetchData() {
    try {
        const response = await fetch('/api/users');
        const data = await response.json();
        displayUsers(data);
    } catch (error) {
        console.error('Error fetching data:', error);
    }
}

// Send data to server
async function saveUser(userData) {
    try {
        const response = await fetch('/api/users', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(userData)
        });
        
        if (response.ok) {
            console.log('User saved successfully');
        }
    } catch (error) {
        console.error('Error saving user:', error);
    }
}

JavaScript Best Practices

  • Use meaningful variable names - userAge instead of x
  • Keep functions small and focused - Each function should do one thing well
  • Use const by default - Only use let when you need to reassign
  • Handle errors gracefully - Use try/catch blocks for error handling
  • Comment your code - Explain why, not just what
  • Use strict mode - Add "use strict"; at the top of your scripts
  • Avoid global variables - Keep variables in appropriate scope
  • Test your code - Use console.log() to debug and verify behavior
